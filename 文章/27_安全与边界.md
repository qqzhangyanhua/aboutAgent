前面的文章把智能体做得越来越聪明——能检索、能记忆、能规划、能调工具。但"聪明"有个副作用：**越聪明的 Agent，能搞的破坏也越大**。

一个只能聊天的 AI 最多回答错了，用户关掉就行。但一个能操作数据库、能发邮件、能下单退款的 Agent——如果它"聪明地"做了一个错误的决定，后果可能是资损、数据泄露甚至法律风险。

这篇文章专门讲**高风险场景下的安全设计模式**——不是泛泛的安全概念，而是实际可落地的代码模式。

## 一、风险分级：不是所有操作都一样危险

### 1.1 四级风险模型

```python
from enum import IntEnum


class RiskLevel(IntEnum):
    LOW = 1       # 只读操作，如查询、搜索
    MEDIUM = 2    # 可逆的写操作，如修改地址
    HIGH = 3      # 难以逆转的操作，如退款、删除
    CRITICAL = 4  # 不可逆 + 高影响，如转账、批量删除


RISK_REGISTRY: dict[str, RiskLevel] = {
    "query_order": RiskLevel.LOW,
    "search_docs": RiskLevel.LOW,
    "track_logistics": RiskLevel.LOW,
    "update_address": RiskLevel.MEDIUM,
    "modify_profile": RiskLevel.MEDIUM,
    "cancel_order": RiskLevel.HIGH,
    "apply_refund": RiskLevel.HIGH,
    "delete_account": RiskLevel.CRITICAL,
    "batch_refund": RiskLevel.CRITICAL,
    "transfer_money": RiskLevel.CRITICAL,
}
```

### 1.2 不同风险等级的处理策略

| 风险等级 | 策略 | 示例 |
|---------|------|------|
| **LOW** | 直接执行，记日志 | 查订单、查物流 |
| **MEDIUM** | 执行前确认参数，记详细日志 | 改地址、改手机号 |
| **HIGH** | 需要用户二次确认 | 退款、取消订单 |
| **CRITICAL** | 需要人工审批 | 批量退款、删除账户 |

## 二、模式一：人工确认（Human-in-the-Loop）

高风险操作不应该由 Agent 自动执行，必须经过人的确认。

```python
import time
import uuid
from dataclasses import dataclass


@dataclass
class PendingAction:
    """待确认的操作"""
    action_id: str
    action_name: str
    params: dict
    risk_level: RiskLevel
    created_at: float
    expires_at: float
    confirmed: bool = False
    rejected: bool = False
    reason: str = ""


class HumanApprovalGate:
    """人工确认门：高风险操作必须经过人的同意"""

    def __init__(self, timeout_seconds: int = 300):
        self.timeout = timeout_seconds
        self.pending: dict[str, PendingAction] = {}

    def request_approval(self, action_name: str, params: dict, risk_level: RiskLevel) -> PendingAction:
        """提交一个待确认的操作"""
        action = PendingAction(
            action_id=str(uuid.uuid4())[:8],
            action_name=action_name,
            params=params,
            risk_level=risk_level,
            created_at=time.time(),
            expires_at=time.time() + self.timeout
        )
        self.pending[action.action_id] = action
        return action

    def approve(self, action_id: str) -> bool:
        """批准操作"""
        action = self.pending.get(action_id)
        if not action or time.time() > action.expires_at:
            return False
        action.confirmed = True
        return True

    def reject(self, action_id: str, reason: str = "") -> bool:
        """拒绝操作"""
        action = self.pending.get(action_id)
        if not action:
            return False
        action.rejected = True
        action.reason = reason
        return True

    def execute_if_approved(self, action_id: str, executor_fn) -> dict:
        """如果已批准，执行操作"""
        action = self.pending.get(action_id)
        if not action:
            return {"error": "操作不存在"}
        if action.rejected:
            return {"error": f"操作已被拒绝：{action.reason}"}
        if not action.confirmed:
            return {"error": "操作尚未被批准"}
        if time.time() > action.expires_at:
            return {"error": "操作已过期"}

        result = executor_fn(action.action_name, action.params)
        return {"success": True, "result": result}


# 示例
gate = HumanApprovalGate(timeout_seconds=300)

# Agent 想执行退款
action = gate.request_approval(
    "apply_refund",
    {"order_id": "12345", "amount": 2999},
    RiskLevel.HIGH
)
print(f"待确认操作：{action.action_name}，金额 ¥{action.params['amount']}")
print(f"操作 ID：{action.action_id}（有效期 5 分钟）")

# 模拟人工批准
gate.approve(action.action_id)
result = gate.execute_if_approved(
    action.action_id,
    lambda name, params: f"退款成功：{params['order_id']}，金额 ¥{params['amount']}"
)
print(f"结果：{result}")
```

## 三、模式二：渐进式权限（Progressive Trust）

新用户/新 Agent 默认只有最低权限，随着信任度提升逐步开放。

```python
class ProgressiveTrust:
    """渐进式信任：根据交互历史动态调整权限"""

    def __init__(self):
        self.trust_scores: dict[str, float] = {}
        self.interaction_counts: dict[str, int] = {}

    def get_trust(self, user_id: str) -> float:
        return self.trust_scores.get(user_id, 0.0)

    def record_interaction(self, user_id: str, success: bool) -> None:
        """记录一次交互，更新信任分"""
        self.interaction_counts[user_id] = self.interaction_counts.get(user_id, 0) + 1
        current = self.trust_scores.get(user_id, 0.0)
        if success:
            self.trust_scores[user_id] = min(1.0, current + 0.05)
        else:
            self.trust_scores[user_id] = max(0.0, current - 0.15)

    def get_allowed_risk_level(self, user_id: str) -> RiskLevel:
        """根据信任分返回允许的最高风险等级"""
        trust = self.get_trust(user_id)
        count = self.interaction_counts.get(user_id, 0)

        if trust >= 0.8 and count >= 50:
            return RiskLevel.HIGH
        elif trust >= 0.5 and count >= 20:
            return RiskLevel.MEDIUM
        else:
            return RiskLevel.LOW

    def check_permission(self, user_id: str, action: str) -> tuple[bool, str]:
        """检查用户是否有权限执行某操作"""
        action_risk = RISK_REGISTRY.get(action, RiskLevel.CRITICAL)
        allowed_risk = self.get_allowed_risk_level(user_id)

        if action_risk <= allowed_risk:
            return True, f"允许（信任分 {self.get_trust(user_id):.2f}）"
        else:
            return False, (
                f"权限不足：{action} 需要 {action_risk.name} 级别，"
                f"当前信任等级只允许 {allowed_risk.name}"
            )


trust = ProgressiveTrust()

# 新用户只能查询
user = "user_001"
for action in ["query_order", "apply_refund", "delete_account"]:
    ok, msg = trust.check_permission(user, action)
    icon = "✅" if ok else "❌"
    print(f"  {icon} {action}: {msg}")

# 积累信任
print("\n--- 积累 30 次成功交互后 ---")
for _ in range(30):
    trust.record_interaction(user, success=True)

for action in ["query_order", "update_address", "apply_refund", "delete_account"]:
    ok, msg = trust.check_permission(user, action)
    icon = "✅" if ok else "❌"
    print(f"  {icon} {action}: {msg}")
```

## 四、模式三：操作限额（Rate & Amount Limits）

```python
from collections import defaultdict


class OperationLimiter:
    """操作限额：防止异常的批量操作"""

    def __init__(self):
        self.limits = {
            "apply_refund": {"max_per_hour": 5, "max_amount_per_day": 10000},
            "cancel_order": {"max_per_hour": 10},
            "delete_account": {"max_per_hour": 1},
        }
        self.hourly_counts: dict[str, dict[str, int]] = defaultdict(lambda: defaultdict(int))
        self.daily_amounts: dict[str, dict[str, float]] = defaultdict(lambda: defaultdict(float))

    def check(self, user_id: str, action: str, amount: float = 0) -> tuple[bool, str]:
        """检查是否超限"""
        if action not in self.limits:
            return True, "无限制"

        limits = self.limits[action]
        key = f"{user_id}:{action}"

        # 频次限制
        max_per_hour = limits.get("max_per_hour")
        if max_per_hour:
            current = self.hourly_counts[key][self._hour_key()]
            if current >= max_per_hour:
                return False, f"超出频次限制：{action} 每小时最多 {max_per_hour} 次"

        # 金额限制
        max_amount = limits.get("max_amount_per_day")
        if max_amount and amount > 0:
            current_amount = self.daily_amounts[key][self._day_key()]
            if current_amount + amount > max_amount:
                return False, f"超出金额限制：今日 {action} 累计不超过 ¥{max_amount}"

        return True, "通过"

    def record(self, user_id: str, action: str, amount: float = 0) -> None:
        """记录一次操作"""
        key = f"{user_id}:{action}"
        self.hourly_counts[key][self._hour_key()] += 1
        if amount > 0:
            self.daily_amounts[key][self._day_key()] += amount

    def _hour_key(self) -> str:
        return time.strftime("%Y%m%d%H")

    def _day_key(self) -> str:
        return time.strftime("%Y%m%d")


limiter = OperationLimiter()

# 正常退款
for i in range(7):
    ok, msg = limiter.check("user_001", "apply_refund", amount=500)
    if ok:
        limiter.record("user_001", "apply_refund", amount=500)
        print(f"  第 {i+1} 次退款：✅ 通过")
    else:
        print(f"  第 {i+1} 次退款：❌ {msg}")
```

## 五、模式四：沙箱执行（Dry Run）

高风险操作先在沙箱里"预演"，确认没问题再真正执行。

```python
class DryRunExecutor:
    """沙箱执行：先预演再真做"""

    def dry_run(self, action: str, params: dict) -> dict:
        """预演：不真正执行，只返回"如果执行会怎样""""
        impacts = {
            "apply_refund": lambda p: {
                "description": f"将退款 ¥{p.get('amount', 0)} 到用户账户",
                "affected_records": [f"订单 {p.get('order_id')}"],
                "reversible": True,
                "estimated_time": "1-3 个工作日"
            },
            "delete_account": lambda p: {
                "description": f"将永久删除用户 {p.get('user_id')} 的所有数据",
                "affected_records": ["用户信息", "订单记录", "收藏夹", "地址簿"],
                "reversible": False,
                "estimated_time": "立即生效"
            },
            "batch_refund": lambda p: {
                "description": f"将批量退款 {len(p.get('order_ids', []))} 笔订单",
                "affected_records": [f"订单 {oid}" for oid in p.get("order_ids", [])],
                "reversible": True,
                "estimated_time": "5-10 分钟"
            },
        }

        impact_fn = impacts.get(action)
        if not impact_fn:
            return {"description": f"执行 {action}", "reversible": True}
        return impact_fn(params)

    def execute_with_preview(self, action: str, params: dict, confirm_fn=None) -> dict:
        """带预览的执行流程"""
        # 1. 预演
        preview = self.dry_run(action, params)
        print(f"  [预演] {preview['description']}")
        print(f"  [影响] {preview['affected_records']}")
        print(f"  [可逆] {'是' if preview['reversible'] else '否'}")

        # 2. 确认
        if confirm_fn and not confirm_fn(preview):
            return {"status": "cancelled", "reason": "用户取消"}

        # 3. 执行
        return {"status": "executed", "preview": preview}


executor = DryRunExecutor()

# 危险操作先预演
executor.execute_with_preview(
    "batch_refund",
    {"order_ids": ["ORD001", "ORD002", "ORD003"], "amount_total": 5000},
    confirm_fn=lambda preview: preview["reversible"]  # 只有可逆操作才自动确认
)
```

## 六、安全设计的六条原则

| # | 原则 | 说明 |
|---|------|------|
| 1 | **最小权限** | Agent 只有完成任务所需的最低权限 |
| 2 | **纵深防御** | 多层检查，不依赖单一防线 |
| 3 | **默认拒绝** | 不在白名单里的操作一律拒绝 |
| 4 | **可审计** | 所有操作都有日志，可事后追溯 |
| 5 | **可逆优先** | 优先使用可撤销的操作方式 |
| 6 | **渐进信任** | 权限随信任度提升，不一步到位 |

## 七、别等出事才加安全

我知道很多开发者对安全机制的态度是"先跑通再说"。能理解——产品经理催着上线，哪有时间搞什么渐进信任。但问题是，Agent 不是普通的 CRUD 应用。**传统应用出 bug，最多数据显示错了；Agent 出 bug，它可能真的去执行了一个操作。**

一个真实场景：某个内部 Agent 连接了 HR 系统，用来帮员工查薪资、请假。开发者偷懒没做操作限额。有个员工测试的时候对 Agent 说"帮我请 100 天假"，Agent 真的提交了 100 天的假期申请到审批系统。虽然审批流拦住了，但 HR 那边收到 100 条审批邮件，差点以为系统被攻击了。

所以我的建议很实际：

- **风险等级表**在设计阶段就画好——你不需要立刻实现所有防护，但你需要知道哪些操作是高风险的
- **操作限额**是成本最低的安全措施——加个计数器，几行代码的事
- **人工确认**只给真正的高风险操作加——退款、删除、修改权限这一类
- **渐进信任和沙箱执行**等业务稳定后再考虑也不迟

核心就一点：别让 Agent 有"无限开火"的能力。再聪明的模型，也不该在没人看的情况下无限制地执行操作。
