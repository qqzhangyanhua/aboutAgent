你的智能体上线了。用户说"它回答得不对"。你问哪里不对？用户说"反正不对"。

然后你打开代码，发现——**你不知道它当时看到了什么上下文、调了哪些工具、模型返回了什么、为什么最终输出了那个回答**。

这就是没有可观测性的代价。

传统软件出了 bug，你看日志、看堆栈、复现一下就能定位。但智能体不一样——**同样的输入，每次输出可能都不一样**（因为大模型有随机性）。你没法复现，只能靠事后回溯。回溯靠什么？靠日志、追踪和监控。

这篇文章讲智能体系统的可观测性：记什么、怎么记、怎么查、怎么告警。

---

## 一、智能体的可观测性为什么特殊？

传统后端服务出了 bug，你打个断点、看个堆栈就能定位。但智能体不行——**同样的输入跑两次，输出可能不同**。你没法复现，更没法打断点（总不能在 GPT-4o 里打断点吧）。

这意味着你对智能体的排错完全依赖**事后回溯**。而回溯的质量取决于你在运行时记录了多少信息。传统软件记函数入参、返回值、报错堆栈就够了。但智能体还需要记：当时的完整 System Prompt 是什么？上下文里塞了哪些检索结果？模型为什么选了工具 A 而不是工具 B？工具调用的顺序是什么？守门人拦截了什么？

少记一个，你在排查问题时就会遇到一个"黑盒"——你知道结果错了，但不知道哪一步出了问题。

## 二、三根支柱：日志、追踪、指标

### 2.1 结构化日志

```python
import json
import time
import uuid
from enum import Enum


class LogLevel(str, Enum):
    DEBUG = "DEBUG"
    INFO = "INFO"
    WARN = "WARN"
    ERROR = "ERROR"


class AgentLogger:
    """智能体专用结构化日志"""

    def __init__(self, service_name: str = "agent-service"):
        self.service_name = service_name

    def log(
        self,
        level: LogLevel,
        event: str,
        trace_id: str,
        data: dict | None = None
    ) -> dict:
        """输出结构化日志"""
        entry = {
            "timestamp": time.strftime("%Y-%m-%dT%H:%M:%S"),
            "level": level.value,
            "service": self.service_name,
            "trace_id": trace_id,
            "event": event,
            "data": data or {}
        }
        print(json.dumps(entry, ensure_ascii=False))
        return entry

    def log_llm_call(
        self,
        trace_id: str,
        model: str,
        messages: list[dict],
        response: str,
        tokens_in: int,
        tokens_out: int,
        latency_ms: int
    ) -> None:
        """记录 LLM 调用"""
        self.log(LogLevel.INFO, "llm_call", trace_id, {
            "model": model,
            "messages_count": len(messages),
            "last_user_msg": next(
                (m["content"][:100] for m in reversed(messages) if m["role"] == "user"),
                ""
            ),
            "response_preview": response[:200],
            "tokens_in": tokens_in,
            "tokens_out": tokens_out,
            "latency_ms": latency_ms
        })

    def log_tool_call(
        self,
        trace_id: str,
        tool_name: str,
        arguments: dict,
        result: str,
        latency_ms: int,
        success: bool
    ) -> None:
        """记录工具调用"""
        self.log(
            LogLevel.INFO if success else LogLevel.ERROR,
            "tool_call",
            trace_id,
            {
                "tool": tool_name,
                "arguments": arguments,
                "result_preview": result[:200],
                "latency_ms": latency_ms,
                "success": success
            }
        )

    def log_guardrail(
        self,
        trace_id: str,
        guard_type: str,
        passed: bool,
        reason: str
    ) -> None:
        """记录守门人检查"""
        self.log(
            LogLevel.INFO if passed else LogLevel.WARN,
            "guardrail_check",
            trace_id,
            {"guard_type": guard_type, "passed": passed, "reason": reason}
        )


# 使用
logger = AgentLogger("customer-service-agent")
trace_id = str(uuid.uuid4())[:8]

logger.log_llm_call(
    trace_id=trace_id,
    model="deepseek-chat",
    messages=[{"role": "user", "content": "我的订单到哪了？"}],
    response="让我帮您查一下，请提供订单号。",
    tokens_in=150,
    tokens_out=30,
    latency_ms=1200
)
```

关键原则：**每条日志都要有 trace_id**。一个用户请求可能触发 3 次 LLM 调用 + 2 次工具调用 + 1 次 RAG 检索——用 trace_id 把它们串起来。

### 2.2 链路追踪

```python
from dataclasses import dataclass, field


@dataclass
class Span:
    """追踪的最小单元"""
    span_id: str
    name: str
    start_time: float
    end_time: float = 0
    parent_id: str | None = None
    attributes: dict = field(default_factory=dict)
    status: str = "ok"

    @property
    def duration_ms(self) -> int:
        return int((self.end_time - self.start_time) * 1000)


class TraceCollector:
    """链路追踪收集器"""

    def __init__(self):
        self.traces: dict[str, list[Span]] = {}

    def start_span(self, trace_id: str, name: str, parent_id: str | None = None) -> Span:
        span = Span(
            span_id=str(uuid.uuid4())[:8],
            name=name,
            start_time=time.time(),
            parent_id=parent_id
        )
        if trace_id not in self.traces:
            self.traces[trace_id] = []
        self.traces[trace_id].append(span)
        return span

    def end_span(self, span: Span, status: str = "ok", attributes: dict | None = None) -> None:
        span.end_time = time.time()
        span.status = status
        if attributes:
            span.attributes.update(attributes)

    def get_trace(self, trace_id: str) -> list[dict]:
        """获取完整的调用链"""
        spans = self.traces.get(trace_id, [])
        return [
            {
                "span_id": s.span_id,
                "name": s.name,
                "parent": s.parent_id,
                "duration_ms": s.duration_ms,
                "status": s.status,
                "attributes": s.attributes
            }
            for s in spans
        ]

    def print_trace(self, trace_id: str) -> None:
        """可视化打印调用链"""
        spans = self.traces.get(trace_id, [])
        if not spans:
            print("未找到追踪数据")
            return

        print(f"\n=== Trace: {trace_id} ===")
        root_spans = [s for s in spans if s.parent_id is None]
        for root in root_spans:
            self._print_span(root, spans, indent=0)

    def _print_span(self, span: Span, all_spans: list[Span], indent: int) -> None:
        prefix = "  " * indent
        status_icon = "✅" if span.status == "ok" else "❌"
        print(f"{prefix}{status_icon} {span.name} ({span.duration_ms}ms)")
        for attr_key, attr_val in span.attributes.items():
            print(f"{prefix}   {attr_key}: {attr_val}")
        children = [s for s in all_spans if s.parent_id == span.span_id]
        for child in children:
            self._print_span(child, all_spans, indent + 1)


# 模拟一次完整请求的追踪
tracer = TraceCollector()
tid = "trace-001"

# 根 Span：整个请求
root = tracer.start_span(tid, "handle_request")

# 子 Span：输入检查
guard_span = tracer.start_span(tid, "input_guard", parent_id=root.span_id)
time.sleep(0.01)
tracer.end_span(guard_span, attributes={"passed": True})

# 子 Span：RAG 检索
rag_span = tracer.start_span(tid, "rag_retrieval", parent_id=root.span_id)
time.sleep(0.05)
tracer.end_span(rag_span, attributes={"docs_found": 3, "top_score": 0.85})

# 子 Span：LLM 调用
llm_span = tracer.start_span(tid, "llm_call", parent_id=root.span_id)
time.sleep(0.1)
tracer.end_span(llm_span, attributes={"model": "deepseek-chat", "tokens": 280})

# 子 Span：工具调用
tool_span = tracer.start_span(tid, "tool_call", parent_id=llm_span.span_id)
time.sleep(0.03)
tracer.end_span(tool_span, attributes={"tool": "query_order", "success": True})

# 子 Span：第二次 LLM 调用
llm2_span = tracer.start_span(tid, "llm_call_2", parent_id=root.span_id)
time.sleep(0.08)
tracer.end_span(llm2_span, attributes={"model": "deepseek-chat", "tokens": 150})

# 子 Span：输出检查
out_span = tracer.start_span(tid, "output_guard", parent_id=root.span_id)
time.sleep(0.01)
tracer.end_span(out_span, attributes={"passed": True})

tracer.end_span(root)
tracer.print_trace(tid)
```

### 2.3 指标与告警

```python
from collections import defaultdict


class MetricsCollector:
    """指标收集器"""

    def __init__(self):
        self.counters: dict[str, int] = defaultdict(int)
        self.gauges: dict[str, float] = {}
        self.histograms: dict[str, list[float]] = defaultdict(list)

    def increment(self, name: str, value: int = 1) -> None:
        self.counters[name] += value

    def set_gauge(self, name: str, value: float) -> None:
        self.gauges[name] = value

    def observe(self, name: str, value: float) -> None:
        self.histograms[name].append(value)

    def get_summary(self) -> dict:
        summary: dict = {}
        summary["counters"] = dict(self.counters)
        summary["gauges"] = dict(self.gauges)
        for name, values in self.histograms.items():
            if values:
                sorted_vals = sorted(values)
                summary[f"histogram_{name}"] = {
                    "count": len(values),
                    "avg": sum(values) / len(values),
                    "p50": sorted_vals[len(values) // 2],
                    "p95": sorted_vals[int(len(values) * 0.95)],
                    "p99": sorted_vals[int(len(values) * 0.99)],
                }
        return summary

    def check_alerts(self) -> list[str]:
        """检查告警规则"""
        alerts: list[str] = []
        # 错误率过高
        total = self.counters.get("requests_total", 0)
        errors = self.counters.get("requests_error", 0)
        if total > 10 and errors / total > 0.1:
            alerts.append(f"错误率过高：{errors}/{total} = {errors/total:.1%}")

        # 延迟过高
        latencies = self.histograms.get("request_latency_ms", [])
        if latencies:
            p95 = sorted(latencies)[int(len(latencies) * 0.95)]
            if p95 > 5000:
                alerts.append(f"P95 延迟过高：{p95:.0f}ms > 5000ms")

        # Token 消耗过快
        daily_tokens = self.counters.get("tokens_total", 0)
        if daily_tokens > 1000000:
            alerts.append(f"日 Token 消耗过高：{daily_tokens:,}")

        return alerts


# 模拟数据
metrics = MetricsCollector()
for _ in range(100):
    metrics.increment("requests_total")
    metrics.observe("request_latency_ms", 800 + (time.time() % 1) * 400)
    metrics.increment("tokens_total", 350)

for _ in range(5):
    metrics.increment("requests_error")

print("=== 指标摘要 ===")
summary = metrics.get_summary()
for key, value in summary.items():
    print(f"  {key}: {value}")

alerts = metrics.check_alerts()
if alerts:
    print("\n=== 告警 ===")
    for alert in alerts:
        print(f"  🚨 {alert}")
```

## 三、智能体专属的可观测数据

除了通用的日志/追踪/指标，智能体还需要记录一些**专属数据**：

| 数据 | 为什么要记 | 用来干什么 |
|------|-----------|-----------|
| 完整的 messages 列表 | 复现模型行为 | 用户投诉时回溯 |
| 模型选择/路由决策 | 理解为什么用了某个模型 | 优化路由策略 |
| 工具调用链 | 知道调了哪些工具、什么顺序 | 分析工具选择准确率 |
| 守门人拦截记录 | 知道拦了什么、为什么拦 | 调整拦截规则 |
| Token 明细 | 知道每次调用花了多少 | 成本优化 |
| 检索结果及评分 | 知道 RAG 检索了什么 | 优化检索质量 |

## 四、一个真实的排查故事

说个事情让你感受一下可观测性的价值。

有一次用户反馈"Agent 回答变慢了，而且经常答非所问"。没有日志的话，你完全无从下手——是模型慢了？是检索慢了？是答非所问因为检索的文档不对？还是 System Prompt 被改坏了？

有了上面这套体系之后，排查过程是这样的：

1. 看指标面板，发现 P95 延迟从 3s 跳到了 12s，从周三开始的
2. 按 trace_id 拉出一条慢请求的完整链路，发现 `retrieval` 这个 Span 花了 9s
3. 看这条请求的日志，发现检索返回了 20 条文档（正常是 5 条），而且相关性评分都很低
4. 回溯周三的变更记录，发现有人把检索的 `top_k` 从 5 改成了 20，同时改了 embedding 模型但没有重建索引

从发现问题到定位根因，15 分钟。如果没有这套东西，大概率得靠猜、靠翻代码 diff、靠在群里问"谁改了什么"，半天能找到算快的。

日志、追踪、指标三个支柱缺哪个都不行：光有日志找不到全链路，光有追踪不知道整体趋势，光有指标知道出事了但不知道出在哪。先把 `trace_id` 串起来，其他的可以慢慢加。
